11 void MEGraphicGpuResourceManager::Initialize(const MEGraphicDevice& device, const UINT numVertexBuffer)
12 {
13 m_numVertexBuffer = numVertexBuffer;
14 D3D12_DESCRIPTOR_HEAP_DESC descHeapDesc = {};
15 descHeapDesc.Flags = D3D12_DESCRIPTOR_HEAP_FLAG_NONE;
16 descHeapDesc.NodeMask = 0;
17 descHeapDesc.NumDescriptors = 2;
18 descHeapDesc.Type = D3D12_DESCRIPTOR_HEAP_TYPE_CBV_SRV_UAV;
19 auto result = device.GetDevice()->CreateDescriptorHeap(&descHeapDesc, IID_PPV_ARGS(m_basicDescHeap.ReleaseAndGetAddressOf()));
20
21 if(FAILED(result))
22 {
23 OutputDebugStringA("DX12 DescriptorHeap Initialize Error.\n");
24 }
25
26 m_descriptorHeapIncrementSize = device.GetDevice()->GetDescriptorHandleIncrementSize(D3D12_DESCRIPTOR_HEAP_TYPE_CBV_SRV_UAV);
27
28 CreateVertexBuffer(4, device);
29 _InitalizeConstantBuffer(device);
30 _CreateCbv(device);
31 }

136 void MEGraphicGpuResourceManager::UploadDataToUploadBuffer(uint8_t * srcData, const size_t rowPitch, const size_t height)
137 {
138 uint8_t* mapforImg = nullptr;
139 auto result = m_uploadBuffer->Map(0, nullptr, (void**)&mapforImg);
140
141 auto srcAddress = srcData;
142 auto GpuRowPitch = _GetAlignmentedSize(rowPitch, D3D12_TEXTURE_DATA_PITCH_ALIGNMENT);
143 for(int y = 0; y < height; ++y)
144 {
145 std::copy_n(srcAddress, GpuRowPitch, mapforImg);
146 srcAddress += rowPitch;
147 mapforImg += GpuRowPitch;
148 }
149 m_uploadBuffer->Unmap(0, nullptr);
150 }

165 void MEGraphicGpuResourceManager::CreateVertexBuffer(const size_t vertexNum, const MEGraphicDevice& device)
166 {
167 D3D12_HEAP_PROPERTIES heapprop = {};
168 heapprop.Type = D3D12_HEAP_TYPE_UPLOAD;
169 heapprop.CPUPageProperty = D3D12_CPU_PAGE_PROPERTY_UNKNOWN;
170 heapprop.MemoryPoolPreference = D3D12_MEMORY_POOL_UNKNOWN;
171
172 D3D12_RESOURCE_DESC resdesc = {};
173 resdesc.Dimension = D3D12_RESOURCE_DIMENSION_BUFFER;
174 resdesc.Width = sizeof(VERTEX_DATA) * vertexNum;
175 resdesc.Height = 1;
176 resdesc.DepthOrArraySize = 1;
177 resdesc.MipLevels = 1;
178 resdesc.Format = DXGI_FORMAT_UNKNOWN;
179 resdesc.SampleDesc.Count = 1;
180 resdesc.Flags = D3D12_RESOURCE_FLAG_NONE;
181 resdesc.Layout = D3D12_TEXTURE_LAYOUT_ROW_MAJOR;
182
183 m_vertexBuffer.resize(m_numVertexBuffer);
184 m_vertexBufferView.resize(m_numVertexBuffer);
185 for(UINT idx = 0; idx < m_numVertexBuffer; ++idx)
186 {
187 auto result = device.GetDevice()->CreateCommittedResource(&heapprop, D3D12_HEAP_FLAG_NONE, &resdesc,
188 D3D12_RESOURCE_STATE_GENERIC_READ, nullptr, IID_PPV_ARGS(m_vertexBuffer[idx].ReleaseAndGetAddressOf()));
189
190 m_vertexBufferView[idx].BufferLocation = m_vertexBuffer[idx]->GetGPUVirtualAddress();
191 m_vertexBufferView[idx].SizeInBytes = static_cast<UINT>(sizeof(VERTEX_DATA) * vertexNum);
192 m_vertexBufferView[idx].StrideInBytes = sizeof(VERTEX_DATA);
193 }
194 }

257 void MEGraphicGpuResourceManager::UploadToGpu(DirectX::TexMetadata& metadata,
258 const size_t rowPitch, const DXGI_FORMAT format, MEGraphicCommandList& cmdList)
259 {
260 D3D12_TEXTURE_COPY_LOCATION src = {};
261 src.pResource = m_uploadBuffer.Get();
262 src.Type = D3D12_TEXTURE_COPY_TYPE_PLACED_FOOTPRINT;
263 src.PlacedFootprint.Offset = 0;
264 src.PlacedFootprint.Footprint.Width = static_cast<UINT>(metadata.width);
265 src.PlacedFootprint.Footprint.Height = static_cast<UINT>(metadata.height);
266 src.PlacedFootprint.Footprint.Depth = static_cast<UINT>(metadata.depth);
267 src.PlacedFootprint.Footprint.RowPitch = static_cast<UINT>(_GetAlignmentedSize(rowPitch, D3D12_TEXTURE_DATA_PITCH_ALIGNMENT));
268 src.PlacedFootprint.Footprint.Format = format;
269
270 D3D12_TEXTURE_COPY_LOCATION dst = {};
271 dst.pResource = m_textureBuffer.Get();
272 dst.Type = D3D12_TEXTURE_COPY_TYPE_SUBRESOURCE_INDEX;
273 dst.SubresourceIndex = 0;
274
275 cmdList.GetCommandList()->CopyTextureRegion(&dst, 0, 0, 0, &src, nullptr);
276
277 _SetBarrierBeforeUploadTexture(cmdList);
278 cmdList.Execute();
279 }

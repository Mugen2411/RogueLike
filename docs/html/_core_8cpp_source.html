62 int MECore::ProcessMessage()
63 {
64 MSG msg = {};
65
66 if(PeekMessage(&msg, nullptr, 0, 0, PM_REMOVE))
67 {
68 TranslateMessage(&msg);
69 DispatchMessage(&msg);
70 }
71
72 if(msg.message == WM_QUIT)
73 {
74 return -1;
75 }
76
77 m_renderTarget.SetBarrierBeforeRender(m_graphicDevice, m_commandList);
78
79 return 0;
80 }

219
220 RECT wrc = { 0, 0, m_windowWidth, m_windowHeight };
221 AdjustWindowRect(&wrc, WS_OVERLAPPEDWINDOW, false);
222
223 m_windowHandle = CreateWindow(m_windowClass.lpszClassName, m_windowTitle.c_str(),
224 WS_OVERLAPPEDWINDOW ^ WS_THICKFRAME, CW_USEDEFAULT, CW_USEDEFAULT,
225 wrc.right - wrc.left, wrc.bottom - wrc.top,
226 nullptr, nullptr, m_windowClass.hInstance, nullptr);
227
228 ShowWindow(m_windowHandle, SW_SHOW);
229 }

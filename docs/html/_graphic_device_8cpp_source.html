23 //DX12 アダプタ列挙
24 std::vector<Microsoft::WRL::ComPtr<IDXGIAdapter>> adapters;
25 Microsoft::WRL::ComPtr<IDXGIAdapter> tmpAdapter = nullptr;
26
27 for(int i = 0; m_dxgiFactory->EnumAdapters(i, &tmpAdapter) != DXGI_ERROR_NOT_FOUND; ++i)
28 {
29 adapters.push_back(tmpAdapter);
30 }
31
32 for(auto adpt : adapters)

34 DXGI_ADAPTER_DESC adesc = {};
35 adpt->GetDesc(&adesc);

37 std::wstring strDesc(adesc.Description);

40 tmpAdapter = adpt;
41 break;
42 }
43 }
44
45 //DX12 デバイス初期化
46 D3D_FEATURE_LEVEL levels[] = {

48 D3D_FEATURE_LEVEL_12_0,
49 D3D_FEATURE_LEVEL_11_1,
50 D3D_FEATURE_LEVEL_11_0
51 };
52
53 D3D_FEATURE_LEVEL featureLevel;
54
55 for(auto lv : levels)

57 if(D3D12CreateDevice(tmpAdapter.Get(), lv, IID_PPV_ARGS(m_device.ReleaseAndGetAddressOf())))
58 {
59 featureLevel = lv;

61 }
62 }

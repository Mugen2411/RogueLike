135 /**********************************************************************//***********************************************************************/
140 void MRPlayer::UpdateOnStand()
141 {
142 auto input = MRInputManager::GetIns();
143 int numPushedButton = 0;
144 float vx = 0.0f;
145 float vy = 0.0f;
146 if (input.GetPushedFrame(MRInputManager::MRKeyCode::RIGHT) > 0)
147 {
148 vx += m_speed;
149 ++numPushedButton;
150 m_isLeft = false;
151 }
152 if (input.GetPushedFrame(MRInputManager::MRKeyCode::LEFT) > 0)
153 {
154 vx -= m_speed;
155 ++numPushedButton;
156 m_isLeft = true;
157 }
158 if (input.GetPushedFrame(MRInputManager::MRKeyCode::DOWN) > 0)
159 {
160 vy += m_speed;
161 ++numPushedButton;
162 }
163 if (input.GetPushedFrame(MRInputManager::MRKeyCode::UP) > 0)
164 {
165 vy -= m_speed;
166 ++numPushedButton;
167 }
168 if (numPushedButton == 0)
169 {
170 m_transform.SetVelocity(0.0f, 0.0f);
171 m_currentAnimation = 0.0f;
172 }
173 else
174 {
175 m_transform.SetVelocity(vx / std::sqrtf(static_cast<float>(numPushedButton)),
176 vy / std::sqrtf(static_cast<float>(numPushedButton)));
177 m_currentAnimation += 0.1f;
178 if (m_currentAnimation > 4.0f) m_currentAnimation -= 4.0f;
179 }
180 if (input.GetPushedFrame(MRInputManager::MRKeyCode::MENU) == 1)
181 {
182 m_frameCount = 0;
183 m_hp.Damage(0.3f);
184 m_transform.SetVelocityWithAngle(6.28f * (rand() % 16) / 16.0f, 4.0f);
185 ChangeState(static_cast<int>(STATE::KNOCKBACKED));
186 return;
187 }
188 }

207 void MRPlayer::UpdateOnKnockbacked()
208 {

210 {
211 m_isLeft = false;
212 }
213 else if (m_transform.GetVelocityX() < -0.1f)
214 {
215 m_isLeft = true;
216 }
217
218 if (m_frameCount > 10)
219 {
220 m_frameCount = 0;
221 ChangeState(static_cast<int>(STATE::STAND));
222 }
223 ++m_frameCount;
224 }

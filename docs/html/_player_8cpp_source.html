185 void MRPlayer::UpdateOnStand()
186 {
187 auto input = MRInputManager::GetIns();
188 int numPushedButton = 0;
189 float vx = 0.0f;
190 float vy = 0.0f;
191 if (input.GetPushedFrame(MRInputManager::MRKeyCode::RIGHT) > 0)
192 {
193 vx += m_speed;
194 ++numPushedButton;
195 m_isLeft = false;
196 }
197 if (input.GetPushedFrame(MRInputManager::MRKeyCode::LEFT) > 0)
198 {
199 vx -= m_speed;
200 ++numPushedButton;
201 m_isLeft = true;
202 }
203 if (input.GetPushedFrame(MRInputManager::MRKeyCode::DOWN) > 0)
204 {
205 vy += m_speed;
206 ++numPushedButton;
207 }
208 if (input.GetPushedFrame(MRInputManager::MRKeyCode::UP) > 0)
209 {
210 vy -= m_speed;
211 ++numPushedButton;
212 }
213 if (numPushedButton == 0)
214 {
215 m_transform.SetVelocity(0.0f, 0.0f);
216 m_animator.Reset();
217 }
218 else
219 {
220 m_transform.SetVelocity(vx / std::sqrtf(static_cast<float>(numPushedButton)),
221 vy / std::sqrtf(static_cast<float>(numPushedButton)));
222 m_animator.Update();
223 }
224 }

238 /**********************************************************************//***********************************************************************/
243 void MRPlayer::UpdateOnKnockbacked()
244 {
245 if (m_transform.GetVelocityX() > 0.1f)
246 {
247 m_isLeft = true;
248 }
249 else if (m_transform.GetVelocityX() < -0.1f)
250 {
251 m_isLeft = false;
252 }
253
254 if (m_stateMachine.GetFrame() > m_knockbackDuration)
255 {
256 m_stateMachine.ChangeState(static_cast<int>(STATE::STAND));
257 }
258 ++m_frameCount;
259 }

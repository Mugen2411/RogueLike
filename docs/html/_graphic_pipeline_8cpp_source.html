20 void MEGraphicPipeline::_ProcessBlobError(const HRESULT result)
21 {
22 if(SUCCEEDED(result))
23 {
24 return;
25 }
26 if(result == HRESULT_FROM_WIN32(ERROR_FILE_NOT_FOUND))
27 {
28 ::OutputDebugStringA("ファイルが見当たりません");
29 return;
30 }
31 std::string errstr;
32 errstr.resize(m_errorBlob->GetBufferSize());
33 std::copy_n(reinterpret_cast<char*>(m_errorBlob->GetBufferPointer()), m_errorBlob->GetBufferSize(), errstr.begin());
34 errstr += '\n';
35 ::OutputDebugStringA(errstr.c_str());
36 }

38 void MEGraphicPipeline::_CreateRootSignarure(const MEGraphicDevice& device)
39 {
40 Microsoft::WRL::ComPtr<ID3DBlob> rootSigBlob = nullptr;
41 //サンプラー
42 D3D12_STATIC_SAMPLER_DESC samplerDesc = {};
43 samplerDesc.AddressU = D3D12_TEXTURE_ADDRESS_MODE_WRAP;
44 samplerDesc.AddressV = D3D12_TEXTURE_ADDRESS_MODE_WRAP;
45 samplerDesc.AddressW = D3D12_TEXTURE_ADDRESS_MODE_WRAP;
46 samplerDesc.BorderColor = D3D12_STATIC_BORDER_COLOR_OPAQUE_BLACK;
47 samplerDesc.Filter = D3D12_FILTER_MIN_MAG_MIP_POINT;
48 samplerDesc.MaxLOD = D3D12_FLOAT32_MAX;
49 samplerDesc.MinLOD = 0.0f;
50 samplerDesc.ShaderVisibility = D3D12_SHADER_VISIBILITY_PIXEL;
51 samplerDesc.ComparisonFunc = D3D12_COMPARISON_FUNC_NEVER;
52 //ディスクリプタレンジ
53 D3D12_DESCRIPTOR_RANGE descTblRange[2] = {};
54 //テクスチャ用
55 descTblRange[0].NumDescriptors = 1;
56 descTblRange[0].RangeType = D3D12_DESCRIPTOR_RANGE_TYPE_SRV;
57 descTblRange[0].BaseShaderRegister = 0;
58 descTblRange[0].OffsetInDescriptorsFromTableStart = D3D12_DESCRIPTOR_RANGE_OFFSET_APPEND;
59 //定数用
60 descTblRange[1].NumDescriptors = 1;
61 descTblRange[1].RangeType = D3D12_DESCRIPTOR_RANGE_TYPE_CBV;
62 descTblRange[1].BaseShaderRegister = 0;
63 descTblRange[1].OffsetInDescriptorsFromTableStart = D3D12_DESCRIPTOR_RANGE_OFFSET_APPEND;
64 //ルートパラメーター
65 D3D12_ROOT_PARAMETER rootparam = {};
66 //テクスチャ用
67 rootparam.ParameterType = D3D12_ROOT_PARAMETER_TYPE_DESCRIPTOR_TABLE;
68 rootparam.ShaderVisibility = D3D12_SHADER_VISIBILITY_ALL;
69 rootparam.DescriptorTable.pDescriptorRanges = descTblRange;
70 rootparam.DescriptorTable.NumDescriptorRanges = 2;
71 //ルートシグネチャ
72 D3D12_ROOT_SIGNATURE_DESC rootSignatureDesc = {};
73 rootSignatureDesc.Flags = D3D12_ROOT_SIGNATURE_FLAG_ALLOW_INPUT_ASSEMBLER_INPUT_LAYOUT;
74 rootSignatureDesc.pParameters = &rootparam;
75 rootSignatureDesc.NumParameters = 1;
76 rootSignatureDesc.pStaticSamplers = &samplerDesc;
77 rootSignatureDesc.NumStaticSamplers = 1;
78 auto result = D3D12SerializeRootSignature(&rootSignatureDesc, D3D_ROOT_SIGNATURE_VERSION_1_0,
79 rootSigBlob.ReleaseAndGetAddressOf(), m_errorBlob.ReleaseAndGetAddressOf());
80 _ProcessBlobError(result);
81 result = device.GetDevice()->CreateRootSignature(0, rootSigBlob->GetBufferPointer(), rootSigBlob->GetBufferSize(),
82 IID_PPV_ARGS(m_rootSignature.ReleaseAndGetAddressOf()));
83 }

85 void MEGraphicPipeline::_LoadShader()
86 {
87 //頂点シェーダー
88 {
89 std::ifstream ifs("BasicVertexShader.cso", std::ios_base::in | std::ios_base::binary);
90 if(ifs.fail())
91 {
92 OutputDebugStringA("VS Load Error\n");
93 return;
94 }
95 ifs.seekg(0, std::ios::end);
96 size_t size = ifs.tellg();
97 ifs.seekg(0, std::ios::beg);
98 m_vsBlob.resize(size);
99 ifs.read(m_vsBlob.data(), size);
100 }
101
102 //ピクセルシェーダー
103 {
104 std::ifstream ifs("BasicPixelShader.cso", std::ios_base::in | std::ios_base::binary);
105 if(ifs.fail())
106 {
107 OutputDebugStringA("PS Load Error\n");
108 return;
109 }
110
111 ifs.seekg(0, std::ios::end);
112 size_t size = ifs.tellg();
113 ifs.seekg(0, std::ios::beg);
114 m_psBlob.resize(size);
115 ifs.read(m_psBlob.data(), size);
116 }
117 }

119 void MEGraphicPipeline::_CreatePipelineState(const MEGraphicDevice& device, const D3D12_INPUT_ELEMENT_DESC inputLayout[], const int layoutSize)
120 {
121 D3D12_GRAPHICS_PIPELINE_STATE_DESC gpipeline = {};
122 //ルートシグネチャ
123 gpipeline.pRootSignature = m_rootSignature.Get();
124 //シェーダー設定
125 gpipeline.VS.pShaderBytecode = m_vsBlob.data();
126 gpipeline.VS.BytecodeLength = m_vsBlob.size();
127 gpipeline.PS.pShaderBytecode = m_psBlob.data();
128 gpipeline.PS.BytecodeLength = m_psBlob.size();
129 //サンプルマスク
130 gpipeline.SampleMask = D3D12_DEFAULT_SAMPLE_MASK;
131 //ラスタライザーステート
132 gpipeline.RasterizerState.MultisampleEnable = false;
133 gpipeline.RasterizerState.CullMode = D3D12_CULL_MODE_NONE;
134 gpipeline.RasterizerState.FillMode = D3D12_FILL_MODE_SOLID;
135 gpipeline.RasterizerState.DepthClipEnable = true;
136 //ブレンドステート
137 gpipeline.BlendState.AlphaToCoverageEnable = false;
138 gpipeline.BlendState.IndependentBlendEnable = false;
139
140 D3D12_RENDER_TARGET_BLEND_DESC renderTargetBlendDesc = {};
141 renderTargetBlendDesc.BlendEnable = true;
142 renderTargetBlendDesc.LogicOpEnable = false;
143 renderTargetBlendDesc.RenderTargetWriteMask = D3D12_COLOR_WRITE_ENABLE_ALL;
144 renderTargetBlendDesc.BlendOp = D3D12_BLEND_OP_ADD;
145 renderTargetBlendDesc.BlendOpAlpha = D3D12_BLEND_OP_ADD;
146 renderTargetBlendDesc.SrcBlend = D3D12_BLEND_SRC_ALPHA;
147 renderTargetBlendDesc.DestBlend = D3D12_BLEND_INV_SRC_ALPHA;
148 renderTargetBlendDesc.SrcBlendAlpha = D3D12_BLEND_ONE;
149 renderTargetBlendDesc.DestBlendAlpha = D3D12_BLEND_ZERO;
150
151 gpipeline.BlendState.RenderTarget[0] = renderTargetBlendDesc;
152 //入力レイアウト
153 gpipeline.InputLayout.pInputElementDescs = inputLayout;
154 gpipeline.InputLayout.NumElements = layoutSize;
155 gpipeline.IBStripCutValue = D3D12_INDEX_BUFFER_STRIP_CUT_VALUE_DISABLED;
156 gpipeline.PrimitiveTopologyType = D3D12_PRIMITIVE_TOPOLOGY_TYPE_TRIANGLE;
157 //レンダーターゲット
158 gpipeline.NumRenderTargets = 1;
159 gpipeline.RTVFormats[0] = DXGI_FORMAT_R8G8B8A8_UNORM;
160 //アンチエイリアス
161 gpipeline.SampleDesc.Count = 1;
162 gpipeline.SampleDesc.Quality = 0;
163
164 auto result = device.GetDevice()->CreateGraphicsPipelineState(&gpipeline, IID_PPV_ARGS(m_pipelineState[0].ReleaseAndGetAddressOf()));
165
166 //加算ブレンド
167 renderTargetBlendDesc.BlendOp = D3D12_BLEND_OP_ADD;
168 renderTargetBlendDesc.SrcBlend = D3D12_BLEND_ONE;
169 renderTargetBlendDesc.DestBlend = D3D12_BLEND_ONE;
170
171 gpipeline.BlendState.RenderTarget[0] = renderTargetBlendDesc;
172
173 result = device.GetDevice()->CreateGraphicsPipelineState(&gpipeline, IID_PPV_ARGS(m_pipelineState[1].ReleaseAndGetAddressOf()));
174
175 //減算ブレンド
176 renderTargetBlendDesc.BlendOp = D3D12_BLEND_OP_SUBTRACT;
177 renderTargetBlendDesc.SrcBlend = D3D12_BLEND_ONE;
178 renderTargetBlendDesc.DestBlend = D3D12_BLEND_ONE;
179
180 gpipeline.BlendState.RenderTarget[0] = renderTargetBlendDesc;
181
182 result = device.GetDevice()->CreateGraphicsPipelineState(&gpipeline, IID_PPV_ARGS(m_pipelineState[2].ReleaseAndGetAddressOf()));
183 }

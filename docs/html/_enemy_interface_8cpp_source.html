54 void MREnemyInterface::HitWithEnemy(MREnemyInterface* rhs)
55 {
56 constexpr float hitSizeRatio = 10.0f;
57 auto& t = rhs->m_transform;
58 float dist2 = m_transform.GetDistance2(t);
59 float maxDist2 = (m_size + rhs->m_size) * (m_size + rhs->m_size);
60 constexpr float inv = 1.0f / 20.0f;
61 if (dist2 < maxDist2)
62 {
63 m_transform.AddVelocityWithAngle(std::atan2f(m_transform.GetY() - t.GetY(), m_transform.GetX() - t.GetX())
64 + m_random.GetRanged(-24, 24) * constants::math::pi2 / 256.0f,
65 static_cast<float>(inv * (m_size + rhs->m_size) * min(maxDist2 / dist2, hitSizeRatio)));
66 }
67 }

75 void MRSlime::_UpdateOnStroll()
76 {
77 if (m_pMapdata->GetDistancePlayer(m_transform) < 32.0f * 8.0f)
78 {
79 m_stateMachine.ChangeState(static_cast<int>(STATE::CHASE));
80 return;
81 }
82 m_animator.Update();
83 if (!Walk() || m_countWalkFrame > 60 || m_stateMachine.GetFrame() == 0)

86 m_countWalkFrame = 0;

88 {
89 m_aliveState = MRAliveState::DISAPPEAR;

91 }

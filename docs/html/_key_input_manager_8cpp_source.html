24 m_currentMousePosX += static_cast<int>(m_mouseSpeedRatio * (mouse.x - center.x));

54 void MEKeyInputManager::Update()
55 {
56 XINPUT_STATE stateXInp;
57 DWORD dwRes = XInputGetState(0, &stateXInp);
58 int numRegistered = static_cast<int>(m_watchingKeyInfos.size());
59 for (int i = 0; i < numRegistered; ++i)
60 {
61 bool flagKB = GetAsyncKeyState(m_watchingKeyInfos[i].codeKeyboard) & 0x8000;
62 bool flagXI = false;
63 if (m_watchingKeyInfos[i].typeGamepadKey == GAMEPAD_KEYTYPE::Buttons)
64 flagXI = stateXInp.Gamepad.wButtons & m_watchingKeyInfos[i].codeGamepad;
65 else if (m_watchingKeyInfos[i].typeGamepadKey == GAMEPAD_KEYTYPE::LTrigger)
66 flagXI = stateXInp.Gamepad.bLeftTrigger > XINPUT_GAMEPAD_TRIGGER_THRESHOLD;
67 else if (m_watchingKeyInfos[i].typeGamepadKey == GAMEPAD_KEYTYPE::RTrigger)
68 flagXI = stateXInp.Gamepad.bRightTrigger > XINPUT_GAMEPAD_TRIGGER_THRESHOLD;
69 else if (m_watchingKeyInfos[i].typeGamepadKey == GAMEPAD_KEYTYPE::ThumbLL)
70 flagXI = stateXInp.Gamepad.sThumbLX < -m_watchingKeyInfos[i].codeGamepad;
71 else if (m_watchingKeyInfos[i].typeGamepadKey == GAMEPAD_KEYTYPE::ThumbLR)
72 flagXI = stateXInp.Gamepad.sThumbLX > m_watchingKeyInfos[i].codeGamepad;
73 else if (m_watchingKeyInfos[i].typeGamepadKey == GAMEPAD_KEYTYPE::ThumbLU)
74 flagXI = stateXInp.Gamepad.sThumbLY < -m_watchingKeyInfos[i].codeGamepad;
75 else if (m_watchingKeyInfos[i].typeGamepadKey == GAMEPAD_KEYTYPE::ThumbLD)
76 flagXI = stateXInp.Gamepad.sThumbLY > m_watchingKeyInfos[i].codeGamepad;
77 else if (m_watchingKeyInfos[i].typeGamepadKey == GAMEPAD_KEYTYPE::ThumbRL)
78 flagXI = stateXInp.Gamepad.sThumbRX < -m_watchingKeyInfos[i].codeGamepad;
79 else if (m_watchingKeyInfos[i].typeGamepadKey == GAMEPAD_KEYTYPE::ThumbRR)
80 flagXI = stateXInp.Gamepad.sThumbRX > m_watchingKeyInfos[i].codeGamepad;
81 else if (m_watchingKeyInfos[i].typeGamepadKey == GAMEPAD_KEYTYPE::ThumbRU)
82 flagXI = stateXInp.Gamepad.sThumbRY < -m_watchingKeyInfos[i].codeGamepad;
83 else if (m_watchingKeyInfos[i].typeGamepadKey == GAMEPAD_KEYTYPE::ThumbRD)
84 flagXI = stateXInp.Gamepad.sThumbRY > m_watchingKeyInfos[i].codeGamepad;
85 if (dwRes != ERROR_SUCCESS)
86 flagXI = false;
87 if (flagKB || flagXI)
88 m_watchingKeyInfos[i].state = ((m_statePrev[i] << 1) ^ 0b011) & 0b011;
89 else
90 m_watchingKeyInfos[i].state = (m_statePrev[i] << 2) & 0b100;
91 m_statePrev[i] = m_watchingKeyInfos[i].state;
92
93 _ProcessMouseInput();
94 }
95 }

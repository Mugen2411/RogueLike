30 m_height = metadata.height / yDivideNum;

32 m_vertices[0] = { { -static_cast<float>(m_width) / 2,-static_cast<float>(m_height) / 2, 0.0f }, { 0.0f, 1.0f } };

42 1.0f / m_yDivideNum * static_cast<int>(index / m_xDivideNum + 1));

44 1.0f / m_yDivideNum * static_cast<int>(index / m_xDivideNum));

49 m_resourceManager.UploadVertexData(index, m_vertices, 4);
50 }

62 void MEImage::DrawGraph(const int x, const int y, const float priority, const int index)
63 {
64 if (index >= m_xDivideNum * m_yDivideNum)
65 {
66 OutputDebugStringA("out of range");

68 }
69 if (x + m_width < 0) return;

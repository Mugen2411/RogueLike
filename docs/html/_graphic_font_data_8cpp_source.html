51 void MEFontData::DrawString(const int x, const int y, const float color[4], const float priority, const std::wstring text)
52 {
53 size_t xOffset = 0;
54 size_t yOffset = 0;
55 for(size_t currentOffset = 0; currentOffset < text.size(); ++currentOffset)
56 {
57 auto character = m_loadedCharacters.find(text[currentOffset]);
58 if(character == m_loadedCharacters.end())
59 {
60 m_loadedCharacters[text[currentOffset]] = MEGraphicCharacterUnit(text[currentOffset], m_oldFont, m_hdc,
61 *m_pDevice, *m_pCmdList, *m_pPipeline, *m_pRenderTarget);
62 }
63 auto& cur = m_loadedCharacters[text[currentOffset]];
64 if(text[currentOffset] == L'\n')
65 {
66 yOffset += cur.m_height;
67 xOffset = 0;
68 }
69 else if (text[currentOffset] == L' ')
70 {
71 xOffset += m_loadedCharacters[text[currentOffset]].m_width;
72 }
73 else
74 {
75 cur.DrawCharacter(x + static_cast<int>(xOffset), y + static_cast<int>(yOffset), color, priority);
76 xOffset += m_loadedCharacters[text[currentOffset]].m_width;
77 }
78 }
79 }

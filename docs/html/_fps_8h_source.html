31 bool Update()
32 {
33 if(count_ == 0)
34 { // 1フレーム目なら時刻を記憶
35 start_time_ = std::chrono::system_clock::now();
36 }
37 if(count_ == cnt_to_sample)
38 { // 60フレーム目なら平均を計算する
39 auto t = std::chrono::system_clock::now();
40 fps_ = 1000.f / (std::chrono::duration_cast<std::chrono::milliseconds>(t - start_time_).count() / (float)cnt_to_sample);
41 count_ = 0;
42 start_time_ = t;
43 }
44 count_++;
45 return true;
46 }

52 }

54 void Wait()
55 {
56 auto tookTime = std::chrono::system_clock::now() - start_time_; //かかった時間
57 long long waitTime = count_ * 1000 / ideal_fps_ - std::chrono::duration_cast<std::chrono::milliseconds>(tookTime).count(); //待つべき時間
58 if(waitTime > 0)
59 {
60 Sleep(static_cast<DWORD>(waitTime)); //待機
61 }
62 }

67 void MEGraphicFontData::DrawString(const int x, const int y, float color[4], const std::wstring text)
68 {
69 size_t xOffset = 0;
70 size_t yOffset = 0;
71 for(size_t currentOffset = 0; currentOffset < text.size(); ++currentOffset)
72 {
73 auto character = m_loadedCharacters.find(text[currentOffset]);
74 if(character == m_loadedCharacters.end())
75 {
76 m_loadedCharacters[text[currentOffset]] = MEGraphicCharacterUnit(text[currentOffset], m_oldFont, m_hdc,
77 *m_pDevice, *m_pCmdList, *m_pPipeline, *m_pRenderTarget);
78 }
79 auto& cur = m_loadedCharacters[text[currentOffset]];
80 if(text[currentOffset] != L'\n')
81 {
82 cur.DrawCharacter(x + static_cast<int>(xOffset), y + static_cast<int>(yOffset), color);
83 xOffset += m_loadedCharacters[text[currentOffset]].m_width;
84 }
85 else
86 {
87 yOffset += cur.m_height;
88 xOffset = 0;
89 }
90 }

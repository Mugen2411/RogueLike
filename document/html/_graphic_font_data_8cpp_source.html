58 void MEGraphicFontData::DrawString(int x, int y, float color[4], std::wstring text)
59 {
60 size_t xOffset = 0;
61 size_t yOffset = 0;
62 for(size_t currentOffset = 0; currentOffset < text.size(); ++currentOffset)
63 {
64 auto character = m_loadedCharacters.find(text[currentOffset]);
65 if(character == m_loadedCharacters.end())
66 {
67 m_loadedCharacters[text[currentOffset]] = MEGraphicCharacterUnit(text[currentOffset], m_oldFont, m_hdc,
68 *m_pDevice, *m_pCmdList, *m_pPipeline, *m_pRenderTarget);
69 }
70 auto& cur = m_loadedCharacters[text[currentOffset]];
71 if(text[currentOffset] != L'\n')
72 {
73 cur.DrawCharacter(x + static_cast<int>(xOffset), y + static_cast<int>(yOffset), color);
74 xOffset += m_loadedCharacters[text[currentOffset]].m_width;
75 }
76 else
77 {
78 yOffset += cur.m_height;
79 xOffset = 0;
80 }
81 }

60 int MECore::ProcessMessage()
61 {
62 MSG msg = {};
63
64 if(PeekMessage(&msg, nullptr, 0, 0, PM_REMOVE))
65 {
66 TranslateMessage(&msg);
67 DispatchMessage(&msg);
68 }
69
70 if(msg.message == WM_QUIT)
71 {
72 return -1;
73 }
74
75 m_renderTarget.SetBarrierBeforeRender(m_graphicDevice, m_commandList);
76

203 void MECore::_CreateWindow()
204 {
205 m_windowClass.cbSize = sizeof(m_windowClass);
206 m_windowClass.lpfnWndProc = WindowProcedure;
207 m_windowClass.lpszClassName = m_windowTitle.c_str();
208 m_windowClass.hInstance = GetModuleHandle(nullptr);
209
210 RegisterClassEx(&m_windowClass);
211
212 RECT wrc = { 0, 0, m_windowWidth, m_windowHeight };
213 AdjustWindowRect(&wrc, WS_OVERLAPPEDWINDOW, false);
214
215 m_windowHandle = CreateWindow(m_windowClass.lpszClassName, m_windowTitle.c_str(),
216 WS_OVERLAPPEDWINDOW, CW_USEDEFAULT, CW_USEDEFAULT,
217 wrc.right - wrc.left, wrc.bottom - wrc.top,
218 nullptr, nullptr, m_windowClass.hInstance, nullptr);
219
220 ShowWindow(m_windowHandle, SW_SHOW);
221 }
222}

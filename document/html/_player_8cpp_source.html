70 void MRPlayer::Update()
71 {
72 auto input = MRInputManager::GetIns();
73 int numPushedButton = 0;
74 float vx = 0.0f;
75 float vy = 0.0f;
76 if (input.GetPushedFrame(MRInputManager::MRKeyCode::RIGHT) > 0)
77 {
78 vx += m_speed;
79 ++numPushedButton;
80 }
81 if (input.GetPushedFrame(MRInputManager::MRKeyCode::LEFT) > 0)
82 {
83 vx -= m_speed;
84 ++numPushedButton;
85 }
86 if (input.GetPushedFrame(MRInputManager::MRKeyCode::DOWN) > 0)
87 {
88 vy += m_speed;
89 ++numPushedButton;
90 }
91 if (input.GetPushedFrame(MRInputManager::MRKeyCode::UP) > 0)
92 {
93 vy -= m_speed;
94 ++numPushedButton;
95 }
96 if (numPushedButton == 0)
97 {
98 m_transform.SetVelocity(0.0f, 0.0f);
99 }
100 else
101 {
102 m_transform.SetVelocity(vx / std::sqrtf(static_cast<float>(numPushedButton)),
103 vy / std::sqrtf(static_cast<float>(numPushedButton)));
104 }
105 }

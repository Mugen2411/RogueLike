55 void MEGraphicCommandList::Execute()
56 {
57 m_cmdList->Close();
58
59 ID3D12CommandList* cmdlists[] = { m_cmdList.Get() };
60 m_cmdQueue->ExecuteCommandLists(1, cmdlists);
61
62 m_cmdQueue->Signal(m_fence.Get(), ++m_fenceVal);
63 if(m_fence->GetCompletedValue() != m_fenceVal)

65 auto event = CreateEvent(nullptr, false, false, nullptr);
66 m_fence->SetEventOnCompletion(m_fenceVal, event);
67 WaitForSingleObject(event, INFINITE);

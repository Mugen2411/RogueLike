30 MEGraphicLoadedImage::MEGraphicLoadedImage(const std::wstring& filepath,
31 MEGraphicDevice& device, size_t xDivideNum, size_t yDivideNum,MEGraphicCommandList& cmdList,
32 MEGraphicPipeline& pipeline, MEGraphicRenderTarget& renderTarget) :
33 m_width(0), m_height(0), m_xDivideNum(xDivideNum), m_yDivideNum(yDivideNum),
34 m_brightness(1.0f, 1.0f, 1.0f, 1.0f), m_pCmdList(&cmdList),
35 m_pPipeline(&pipeline), m_pRenderTarget(&renderTarget)
36 {
37 DirectX::TexMetadata metadata = {};
38 DirectX::ScratchImage scratchImg = {};
39 DirectX::Image const* img = {};
40
41 DirectX::LoadFromWICFile(filepath.c_str(), DirectX::WIC_FLAGS_NONE, &metadata, scratchImg);
42 img = scratchImg.GetImage(0, 0, 0);
43
44 m_width = metadata.width / xDivideNum;
45 m_height = metadata.height / yDivideNum;
46
47 m_vertices[0] = {{ -static_cast<float>(m_width) / 2,-static_cast<float>(m_height) / 2, 0.0f }, { 0.0f, 1.0f }};
48 m_vertices[1] = { {-static_cast<float>(m_width) / 2, static_cast<float>(m_height) / 2, 0.0f},{0.0f, 0.0f} };
49 m_vertices[2] = { { static_cast<float>(m_width) / 2,-static_cast<float>(m_height) / 2, 0.0f},{1.0f, 1.0f} };
50 m_vertices[3] = { { static_cast<float>(m_width) / 2, static_cast<float>(m_height) / 2, 0.0f},{1.0f, 0.0f} };
51
52 m_resourceManager.Initialize(device);
53 m_resourceManager.CreateTextureBuffer(metadata, device);
54 m_resourceManager.CreateSrv(img->format, device);
55
56 m_resourceManager.ResetUploadBuffer(img->rowPitch, img->height, device);
57 m_resourceManager.UploadDataToUploadBuffer(img->pixels, img->rowPitch, img->height);
58 m_resourceManager.UploadToGpu(metadata, img->rowPitch, img->format, *m_pCmdList);
59 }

68 void MEGraphicLoadedImage::DrawGraph(int x, int y, int index)
69 {
70 if(index >= m_xDivideNum * m_yDivideNum)
71 {
72 OutputDebugStringA("out of range");
73 return;
74 }
75 m_vertices[0].uv = DirectX::XMFLOAT2(1.0f / m_xDivideNum * (index % m_xDivideNum),
76 1.0f / m_yDivideNum * static_cast<int>(index / m_xDivideNum + 1));
77 m_vertices[1].uv = DirectX::XMFLOAT2(1.0f / m_xDivideNum * (index % m_xDivideNum),
78 1.0f / m_yDivideNum * static_cast<int>(index / m_xDivideNum));
79 m_vertices[2].uv = DirectX::XMFLOAT2(1.0f / m_xDivideNum * (index % m_xDivideNum + 1),
80 1.0f / m_yDivideNum * static_cast<int>(index / m_xDivideNum + 1));
81 m_vertices[3].uv = DirectX::XMFLOAT2(1.0f / m_xDivideNum * (index % m_xDivideNum + 1),
82 1.0f / m_yDivideNum * static_cast<int>(index / m_xDivideNum));
83 CONSTANT_DATA constData = {};
84 constData.scaleMatrix = DirectX::XMMatrixScaling(2.0f / MECore::GetIns().GetWindowWidth(),
85 2.0f / MECore::GetIns().GetWindowHeight(), 1.0f);
86 constData.moveMatrix = DirectX::XMMatrixTranslation(
87 static_cast<float>(x - MECore::GetIns().GetWindowWidth() / 2) / MECore::GetIns().GetWindowWidth() * 2,
88 static_cast<float>(-y + MECore::GetIns().GetWindowHeight() / 2) / MECore::GetIns().GetWindowHeight() * 2, 0.0f);
89 constData.rotateMatrix = DirectX::XMMatrixIdentity();
90 constData.brightness = m_brightness;
91
92 m_pRenderTarget->SetRenderBaseCommand(*m_pCmdList);
93 m_pPipeline->SetPipelineState(0, *m_pCmdList);
94 m_resourceManager.SetGpuResource(*m_pCmdList);
95
96 m_pCmdList->GetCommandList()->IASetPrimitiveTopology(D3D_PRIMITIVE_TOPOLOGY_TRIANGLESTRIP);
97 m_resourceManager.UploadVertexData(m_vertices, _countof(m_vertices));
98 m_resourceManager.UploadConstantData(constData);
99 m_resourceManager.SetRenderCommand(*m_pCmdList);
100
101 m_pCmdList->Execute();
102 }

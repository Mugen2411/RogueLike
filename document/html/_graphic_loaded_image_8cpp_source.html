66 void MEGraphicLoadedImage::DrawGraph(int x, int y, int index)
67 {
68 if(index >= m_xDivideNum * m_yDivideNum)
69 {
70 OutputDebugStringA("out of range");
71 return;
72 }
73 m_vertices[0].uv = DirectX::XMFLOAT2(1.0f / m_xDivideNum * (index % m_xDivideNum),
74 1.0f / m_yDivideNum * static_cast<int>(index / m_xDivideNum + 1));
75 m_vertices[1].uv = DirectX::XMFLOAT2(1.0f / m_xDivideNum * (index % m_xDivideNum),
76 1.0f / m_yDivideNum * static_cast<int>(index / m_xDivideNum));
77 m_vertices[2].uv = DirectX::XMFLOAT2(1.0f / m_xDivideNum * (index % m_xDivideNum + 1),
78 1.0f / m_yDivideNum * static_cast<int>(index / m_xDivideNum + 1));
79 m_vertices[3].uv = DirectX::XMFLOAT2(1.0f / m_xDivideNum * (index % m_xDivideNum + 1),
80 1.0f / m_yDivideNum * static_cast<int>(index / m_xDivideNum));
81 CONSTANT_DATA constData = {};
82 constData.scaleMatrix = DirectX::XMMatrixScaling(2.0f / MECore::GetIns().m_windowWidth,
83 2.0f / MECore::GetIns().m_windowHeight, 1.0f);
84 constData.moveMatrix = DirectX::XMMatrixTranslation(
85 static_cast<float>(x - MECore::GetIns().m_windowWidth / 2) / MECore::GetIns().m_windowWidth * 2,
86 static_cast<float>(-y + MECore::GetIns().m_windowHeight / 2) / MECore::GetIns().m_windowHeight * 2, 0.0f);
87 constData.rotateMatrix = DirectX::XMMatrixIdentity();
88 constData.brightness = m_brightness;
89
90 m_renderTarget->SetRenderBaseCommand(*m_cmdList);
91 m_pipeline->SetPipelineState(0, *m_cmdList);
92 m_resourceManager->SetGpuResource(m_index, *m_cmdList);
93
94 m_cmdList->GetCommandList()->IASetPrimitiveTopology(D3D_PRIMITIVE_TOPOLOGY_TRIANGLESTRIP);
95 m_resourceManager->UploadVertexData(m_vertices, _countof(m_vertices), *m_cmdList);
96 m_resourceManager->UploadConstantData(m_index, constData, *m_cmdList);
97 m_resourceManager->SetRenderCommand(*m_cmdList);
98
99 m_cmdList->Execute();
100 }

31 MEImage::MEImage(const std::wstring& filepath,
32 MEGraphicDevice& device, size_t xDivideNum, size_t yDivideNum,MEGraphicCommandList& cmdList,
33 MEGraphicPipeline& pipeline, MEGraphicRenderTarget& renderTarget) :
34 m_width(0), m_height(0), m_xDivideNum(xDivideNum), m_yDivideNum(yDivideNum),
35 m_brightness(1.0f, 1.0f, 1.0f, 1.0f), m_pCmdList(&cmdList),
36 m_pPipeline(&pipeline), m_pRenderTarget(&renderTarget), m_blendType(BLEND_TYPE::ALPHA)
37 {
38 DirectX::TexMetadata metadata = {};
39 DirectX::ScratchImage scratchImg = {};
40 DirectX::Image const* img = {};
41
42 DirectX::LoadFromWICFile(filepath.c_str(), DirectX::WIC_FLAGS_NONE, &metadata, scratchImg);
43 img = scratchImg.GetImage(0, 0, 0);
44
45 m_width = metadata.width / xDivideNum;
46 m_height = metadata.height / yDivideNum;
47
48 m_vertices[0] = {{ -static_cast<float>(m_width) / 2,-static_cast<float>(m_height) / 2, 0.0f }, { 0.0f, 1.0f }};
49 m_vertices[1] = { {-static_cast<float>(m_width) / 2, static_cast<float>(m_height) / 2, 0.0f},{0.0f, 0.0f} };
50 m_vertices[2] = { { static_cast<float>(m_width) / 2,-static_cast<float>(m_height) / 2, 0.0f},{1.0f, 1.0f} };
51 m_vertices[3] = { { static_cast<float>(m_width) / 2, static_cast<float>(m_height) / 2, 0.0f},{1.0f, 0.0f} };
52
53 m_resourceManager.Initialize(device, static_cast<UINT>(xDivideNum * yDivideNum));
54
55 for(int index = 0; index < xDivideNum * yDivideNum; ++index)
56 {
57 m_vertices[0].uv = DirectX::XMFLOAT2(1.0f / m_xDivideNum * (index % m_xDivideNum),
58 1.0f / m_yDivideNum * static_cast<int>(index / m_xDivideNum + 1));
59 m_vertices[1].uv = DirectX::XMFLOAT2(1.0f / m_xDivideNum * (index % m_xDivideNum),
60 1.0f / m_yDivideNum * static_cast<int>(index / m_xDivideNum));
61 m_vertices[2].uv = DirectX::XMFLOAT2(1.0f / m_xDivideNum * (index % m_xDivideNum + 1),
62 1.0f / m_yDivideNum * static_cast<int>(index / m_xDivideNum + 1));
63 m_vertices[3].uv = DirectX::XMFLOAT2(1.0f / m_xDivideNum * (index % m_xDivideNum + 1),
64 1.0f / m_yDivideNum * static_cast<int>(index / m_xDivideNum));
65 m_resourceManager.UploadVertexData(index, m_vertices, 4);
66 }
67
68 m_resourceManager.CreateTextureBuffer(metadata, device);
69 m_resourceManager.CreateSrv(img->format, device);
70
71 m_resourceManager.UploadByCpu(img->pixels, img->rowPitch, img->height);

85 void MEImage::DrawGraph(int x, int y, float priority, int index)
86 {
87 if(index >= m_xDivideNum * m_yDivideNum)
88 {
89 OutputDebugStringA("out of range");
90 return;
91 }
92 if (x + m_width < 0) return;
93 if (x - m_width > MECore::GetIns().GetWindowWidth()) return;
94 if (y + m_height < 0) return;
95 if (y - m_height > MECore::GetIns().GetWindowHeight()) return;
96 CONSTANT_DATA constData = {};
97 constData.scaleMatrix = DirectX::XMMatrixScaling(2.0f / MECore::GetIns().GetWindowWidth(),
98 2.0f / MECore::GetIns().GetWindowHeight(), 1.0f);
99 constData.moveMatrix = DirectX::XMMatrixTranslation(
100 static_cast<float>(x - MECore::GetIns().GetWindowWidth() / 2) / MECore::GetIns().GetWindowWidth() * 2,
101 static_cast<float>(-y + MECore::GetIns().GetWindowHeight() / 2) / MECore::GetIns().GetWindowHeight() * 2, 0.0f);
102 constData.rotateMatrix = DirectX::XMMatrixIdentity();
103 constData.brightness = m_brightness;
104
105 MEGraphicRenderQueue::ReserveRender(m_resourceManager.GetVertexBufferView(index), constData,
106 &m_resourceManager, m_blendType, priority, m_pCmdList, m_pPipeline, m_pRenderTarget);
107 }

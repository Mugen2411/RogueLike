195 void MEGraphicGpuResourceManager::UploadDataToUploadBuffer(uint8_t * srcData, const size_t rowPitch, const size_t height)
196 {
197 uint8_t* mapforImg = nullptr;
198 auto result = m_uploadBuffer->Map(0, nullptr, (void**)&mapforImg);
199
200 auto srcAddress = srcData;
201 auto GpuRowPitch = _GetAlignmentedSize(rowPitch, D3D12_TEXTURE_DATA_PITCH_ALIGNMENT);
202 for(int y = 0; y < height; ++y)
203 {
204 std::copy_n(srcAddress, GpuRowPitch, mapforImg);
205 srcAddress += rowPitch;
206 mapforImg += GpuRowPitch;
207 }
208 m_uploadBuffer->Unmap(0, nullptr);
209 }

235 void MEGraphicGpuResourceManager::CreateVertexBuffer(size_t vertexNum, const MEGraphicDevice& device)
236 {
237 D3D12_HEAP_PROPERTIES heapprop = {};
238 heapprop.Type = D3D12_HEAP_TYPE_UPLOAD;
239 heapprop.CPUPageProperty = D3D12_CPU_PAGE_PROPERTY_UNKNOWN;
240 heapprop.MemoryPoolPreference = D3D12_MEMORY_POOL_UNKNOWN;
241
242 D3D12_RESOURCE_DESC resdesc = {};
243 resdesc.Dimension = D3D12_RESOURCE_DIMENSION_BUFFER;
244 resdesc.Width = sizeof(VERTEX_DATA) * vertexNum;
245 resdesc.Height = 1;
246 resdesc.DepthOrArraySize = 1;
247 resdesc.MipLevels = 1;
248 resdesc.Format = DXGI_FORMAT_UNKNOWN;
249 resdesc.SampleDesc.Count = 1;
250 resdesc.Flags = D3D12_RESOURCE_FLAG_NONE;
251 resdesc.Layout = D3D12_TEXTURE_LAYOUT_ROW_MAJOR;
252
253 m_vertexBuffer.resize(m_numVertexBuffer);
254 m_vertexBufferView.resize(m_numVertexBuffer);
255 for(UINT idx = 0; idx < m_numVertexBuffer; ++idx)
256 {
257 auto result = device.GetDevice()->CreateCommittedResource(&heapprop, D3D12_HEAP_FLAG_NONE, &resdesc,
258 D3D12_RESOURCE_STATE_GENERIC_READ, nullptr, IID_PPV_ARGS(m_vertexBuffer[idx].ReleaseAndGetAddressOf()));
259
260 m_vertexBufferView[idx].BufferLocation = m_vertexBuffer[idx]->GetGPUVirtualAddress();
261 m_vertexBufferView[idx].SizeInBytes = static_cast<UINT>(sizeof(VERTEX_DATA) * vertexNum);
262 m_vertexBufferView[idx].StrideInBytes = sizeof(VERTEX_DATA);
263 }

347 MEGraphicCommandList& cmdList)
348 {
349 D3D12_TEXTURE_COPY_LOCATION src = {};
350 src.pResource = m_uploadBuffer.Get();

352 src.PlacedFootprint.Offset = 0;
353 src.PlacedFootprint.Footprint.Width = static_cast<UINT>(metadata.width);
354 src.PlacedFootprint.Footprint.Height = static_cast<UINT>(metadata.height);
355 src.PlacedFootprint.Footprint.Depth = static_cast<UINT>(metadata.depth);
356 src.PlacedFootprint.Footprint.RowPitch = static_cast<UINT>(_GetAlignmentedSize(rowPitch, D3D12_TEXTURE_DATA_PITCH_ALIGNMENT));
357 src.PlacedFootprint.Footprint.Format = format;
358
359 D3D12_TEXTURE_COPY_LOCATION dst = {};

361 dst.Type = D3D12_TEXTURE_COPY_TYPE_SUBRESOURCE_INDEX;
362 dst.SubresourceIndex = 0;
363

365
366 _SetBarrierBeforeUploadTexture(cmdList);
367 cmdList.Execute();

21 void MEGraphicGpuResourceManager::Initialize(const MEGraphicDevice& device)
22 {
23 D3D12_DESCRIPTOR_HEAP_DESC descHeapDesc = {};
24 descHeapDesc.Flags = D3D12_DESCRIPTOR_HEAP_FLAG_SHADER_VISIBLE;
25 descHeapDesc.NodeMask = 0;
26 descHeapDesc.NumDescriptors = 2;
27 descHeapDesc.Type = D3D12_DESCRIPTOR_HEAP_TYPE_CBV_SRV_UAV;
28 auto result = device.GetDevice()->CreateDescriptorHeap(&descHeapDesc, IID_PPV_ARGS(m_basicDescHeap.ReleaseAndGetAddressOf()));
29
30 if(FAILED(result))
31 {
32 OutputDebugStringA("DX12 DescriptorHeap Initialize Error.\n");
33 }
34
35 m_descriptorHeapIncrementSize = device.GetDevice()->GetDescriptorHandleIncrementSize(D3D12_DESCRIPTOR_HEAP_TYPE_CBV_SRV_UAV);
36
37 _CreateVertexBuffer(4, device);
38
39 _InitalizeConstantBuffer(device);
40 _CreateCbv(device);
41 }

160 resdesc.Flags = D3D12_RESOURCE_FLAG_NONE;
161 resdesc.Layout = D3D12_TEXTURE_LAYOUT_UNKNOWN;
162
163 auto result = device.GetDevice()->CreateCommittedResource(&heapprop, D3D12_HEAP_FLAG_NONE, &resdesc,
164 D3D12_RESOURCE_STATE_COPY_DEST, nullptr, IID_PPV_ARGS(m_textureBuffer.ReleaseAndGetAddressOf()));

206 void MEGraphicGpuResourceManager::UploadDataToUploadBuffer(uint8_t * srcData, const size_t rowPitch, const size_t height)
207 {
208 uint8_t* mapforImg = nullptr;
209 auto result = m_uploadBuffer->Map(0, nullptr, (void**)&mapforImg);
210
211 auto srcAddress = srcData;
212 auto GpuRowPitch = _GetAlignmentedSize(rowPitch, D3D12_TEXTURE_DATA_PITCH_ALIGNMENT);
213 for(int y = 0; y < height; ++y)
214 {
215 std::copy_n(srcAddress, GpuRowPitch, mapforImg);
216 srcAddress += rowPitch;
217 mapforImg += GpuRowPitch;
218 }
219 m_uploadBuffer->Unmap(0, nullptr);
220 }

280 void MEGraphicGpuResourceManager::UploadToGpu(DirectX::TexMetadata& metadata, size_t rowPitch, DXGI_FORMAT format,
281 MEGraphicCommandList& cmdList)
282 {
283 D3D12_TEXTURE_COPY_LOCATION src = {};
284 src.pResource = m_uploadBuffer.Get();
285 src.Type = D3D12_TEXTURE_COPY_TYPE_PLACED_FOOTPRINT;
286 src.PlacedFootprint.Offset = 0;
287 src.PlacedFootprint.Footprint.Width = static_cast<UINT>(metadata.width);
288 src.PlacedFootprint.Footprint.Height = static_cast<UINT>(metadata.height);
289 src.PlacedFootprint.Footprint.Depth = static_cast<UINT>(metadata.depth);
290 src.PlacedFootprint.Footprint.RowPitch = static_cast<UINT>(_GetAlignmentedSize(rowPitch, D3D12_TEXTURE_DATA_PITCH_ALIGNMENT));
291 src.PlacedFootprint.Footprint.Format = format;
292
293 D3D12_TEXTURE_COPY_LOCATION dst = {};
294 dst.pResource = m_textureBuffer.Get();
295 dst.Type = D3D12_TEXTURE_COPY_TYPE_SUBRESOURCE_INDEX;
296 dst.SubresourceIndex = 0;
297
298 cmdList.GetCommandList()->CopyTextureRegion(&dst, 0, 0, 0, &src, nullptr);
299
300 _SetBarrierBeforeUploadTexture(cmdList);
301 cmdList.Execute();
302 }

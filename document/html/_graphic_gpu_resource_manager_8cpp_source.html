21 void MEGraphicGpuResourceManager::Initialize(const MEGraphicDevice& device)
22 {
23 D3D12_DESCRIPTOR_HEAP_DESC descHeapDesc = {};
24 descHeapDesc.Flags = D3D12_DESCRIPTOR_HEAP_FLAG_SHADER_VISIBLE;
25 descHeapDesc.NodeMask = 0;
26 descHeapDesc.NumDescriptors = m_maxResourceView * 2;
27 descHeapDesc.Type = D3D12_DESCRIPTOR_HEAP_TYPE_CBV_SRV_UAV;
28 auto result = device.GetDevice()->CreateDescriptorHeap(&descHeapDesc, IID_PPV_ARGS(m_basicDescHeap.ReleaseAndGetAddressOf()));
29
30 if(FAILED(result))
31 {
32 OutputDebugStringA("DX12 DescriptorHeap Initialize Error.\n");
33 }
34
35 m_descriptorHeapIncrementSize = device.GetDevice()->GetDescriptorHandleIncrementSize(D3D12_DESCRIPTOR_HEAP_TYPE_CBV_SRV_UAV);
36
37 _CreateVertexBuffer(4, device);
38 }

49 MEGraphicLoadedImage MEGraphicGpuResourceManager::LoadGraph(const std::wstring& filepath,
50 const MEGraphicDevice& device, MEGraphicCommandList& cmdList,
51 MEGraphicPipeline& pipeline, MEGraphicRenderTarget& renderTarget)
52 {
53 auto index = _GetShaderResourceIndex();
54 _InitalizeConstantBuffer(index, device);
55
56 DirectX::TexMetadata metadata = {};
57 DirectX::ScratchImage scratchImg = {};
58 DirectX::Image const* img = {};
59
60 DirectX::LoadFromWICFile(filepath.c_str(), DirectX::WIC_FLAGS_NONE, &metadata, scratchImg);
61 img = scratchImg.GetImage(0, 0, 0);
62
63 _CreateTextureBuffer(index, metadata, device);
64 _CreateCbv(index, device);
65 _CreateSrv(index, img->format, device);
66
67 _ResetUploadBuffer(img->rowPitch, img->height, device);
68 _UploadDataToUploadBuffer(img->pixels, img->rowPitch, img->height);
69 _UploadToGpu(index, metadata, img->rowPitch, img->format, cmdList);
70
71 MEGraphicLoadedImage ret(index, img->width, img->height, &cmdList, this, &pipeline, &renderTarget);
72
73 return ret;
74 }

192 void MEGraphicGpuResourceManager::_CreateSrv(uint32_t index, const DXGI_FORMAT format,
193 const MEGraphicDevice& device)
194 {
195 auto basicHeapHandle = m_basicDescHeap->GetCPUDescriptorHandleForHeapStart();
196 basicHeapHandle.ptr += m_descriptorHeapIncrementSize * (index * 2);
197
198 D3D12_SHADER_RESOURCE_VIEW_DESC srvDesc = {};
199 srvDesc.Format = format;
200 srvDesc.Shader4ComponentMapping = D3D12_DEFAULT_SHADER_4_COMPONENT_MAPPING;
201 srvDesc.ViewDimension = D3D12_SRV_DIMENSION_TEXTURE2D;
202 srvDesc.Texture2D.MipLevels = 1;
203
204 device.GetDevice()->CreateShaderResourceView(m_textureBuffers[index].Get(), &srvDesc, basicHeapHandle);
205 }

252 resdesc.Layout = D3D12_TEXTURE_LAYOUT_UNKNOWN;
253

255 D3D12_RESOURCE_STATE_COPY_DEST, nullptr, IID_PPV_ARGS(m_textureBuffers[index].ReleaseAndGetAddressOf()));

297 void MEGraphicGpuResourceManager::_UploadDataToUploadBuffer(uint8_t * srcData, const size_t rowPitch, const size_t height)
298 {
299 uint8_t* mapforImg = nullptr;
300 auto result = m_uploadBuffer->Map(0, nullptr, (void**)&mapforImg);
301
302 auto srcAddress = srcData;
303 auto GpuRowPitch = _GetAlignmentedSize(rowPitch, D3D12_TEXTURE_DATA_PITCH_ALIGNMENT);
304 for(int y = 0; y < height; ++y)
305 {
306 std::copy_n(srcAddress, GpuRowPitch, mapforImg);
307 srcAddress += rowPitch;
308 mapforImg += GpuRowPitch;
309 }
310 m_uploadBuffer->Unmap(0, nullptr);
311 }

373 void MEGraphicGpuResourceManager::_UploadToGpu(uint32_t index, DirectX::TexMetadata& metadata, size_t rowPitch, DXGI_FORMAT format,
374 MEGraphicCommandList& cmdList)
375 {
376 D3D12_TEXTURE_COPY_LOCATION src = {};
377 src.pResource = m_uploadBuffer.Get();
378 src.Type = D3D12_TEXTURE_COPY_TYPE_PLACED_FOOTPRINT;
379 src.PlacedFootprint.Offset = 0;
380 src.PlacedFootprint.Footprint.Width = static_cast<UINT>(metadata.width);
381 src.PlacedFootprint.Footprint.Height = static_cast<UINT>(metadata.height);
382 src.PlacedFootprint.Footprint.Depth = static_cast<UINT>(metadata.depth);
383 src.PlacedFootprint.Footprint.RowPitch = static_cast<UINT>(_GetAlignmentedSize(rowPitch, D3D12_TEXTURE_DATA_PITCH_ALIGNMENT));
384 src.PlacedFootprint.Footprint.Format = format;
385
386 D3D12_TEXTURE_COPY_LOCATION dst = {};
387 dst.pResource = m_textureBuffers[index].Get();
388 dst.Type = D3D12_TEXTURE_COPY_TYPE_SUBRESOURCE_INDEX;
389 dst.SubresourceIndex = 0;
390
391 cmdList.GetCommandList()->CopyTextureRegion(&dst, 0, 0, 0, &src, nullptr);
392
393 _SetBarrierBeforeUploadTexture(index, cmdList);
394 cmdList.Execute();
395 }

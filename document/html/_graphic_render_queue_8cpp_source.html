72 void MEGraphicRenderQueue::RenderAll(MEGraphicCommandList& cmdList, MEGraphicPipeline& pipeline, MEGraphicRenderTarget& renderTarget)
73 {
74 for (auto& res : m_reserveList)
75 {
76 m_reservePointerList.push_back(&res);
77 }
78 std::sort(m_reservePointerList.begin(), m_reservePointerList.end(), [](const RENDER_DATA* lhs, const RENDER_DATA* rhs)
79 {
80 if(lhs->priority > rhs->priority) return true;
81 if(lhs->priority < rhs->priority) return false;
82 if(lhs->order < rhs->order) return true;
83 return false;
84 });
85 do
86 {
87 auto cpuHandle = m_constantDescHeap->GetCPUDescriptorHandleForHeapStart();
88 const size_t processNum = min(m_reservePointerList.size(), m_maxReserve);
89 auto itr = m_reservePointerList.begin();
90 for(int idx = 0; idx < processNum; ++idx, ++itr)
91 {
92 m_pDevice->GetDevice()->CopyDescriptorsSimple(1, cpuHandle,
93 (*itr)->textureHeap->GetTextureHeap()->GetCPUDescriptorHandleForHeapStart(), D3D12_DESCRIPTOR_HEAP_TYPE_CBV_SRV_UAV);
94 cpuHandle.ptr += m_descriptorHeapIncrementSize * 2;
95 *m_pMapMatrix[idx] = (*itr)->constData;
96 }
97
98 int beforeBlendType = -1;
99 auto handle = m_constantDescHeap->GetGPUDescriptorHandleForHeapStart();
100 itr = m_reservePointerList.begin();
101 renderTarget.SetRenderBaseCommand(cmdList);
102 for(int idx = 0; idx < processNum; ++idx, ++itr)
103 {
104 if (beforeBlendType != (*itr)->blendType)
105 {
106 pipeline.SetPipelineState((*itr)->blendType, cmdList);
107 beforeBlendType = (*itr)->blendType;
108 }
109 cmdList.GetCommandList()->SetDescriptorHeaps(1, m_constantDescHeap.GetAddressOf());
110 cmdList.GetCommandList()->SetGraphicsRootDescriptorTable(0, handle);
111
112 cmdList.GetCommandList()->IASetPrimitiveTopology(D3D_PRIMITIVE_TOPOLOGY_TRIANGLESTRIP);
113 cmdList.GetCommandList()->IASetVertexBuffers(0, 1, &(*itr)->vertexBufferView);
114 cmdList.GetCommandList()->DrawInstanced(4, 1, 0, 0);
115
116 handle.ptr += m_descriptorHeapIncrementSize * 2;
117 }
118 m_reservePointerList.erase(m_reservePointerList.begin(), m_reservePointerList.begin() + processNum);
119 cmdList.Execute();
120 } while(!m_reservePointerList.empty());
121 m_reserveList.clear();
122 }

130 void MEGraphicRenderQueue::_CreateCbv(uint32_t index, const MEGraphicDevice& device)
131 {
132 D3D12_HEAP_PROPERTIES heapprop = {};
133 heapprop.Type = D3D12_HEAP_TYPE_UPLOAD;
134 heapprop.CPUPageProperty = D3D12_CPU_PAGE_PROPERTY_UNKNOWN;
135 heapprop.MemoryPoolPreference = D3D12_MEMORY_POOL_UNKNOWN;
136 heapprop.CreationNodeMask = 0;
137 heapprop.VisibleNodeMask = 0;
138
139 D3D12_RESOURCE_DESC resdesc = {};
140 resdesc.Dimension = D3D12_RESOURCE_DIMENSION_BUFFER;
141 resdesc.Width = (sizeof(CONSTANT_DATA) + 0xFF) & ~0xFF;
142 resdesc.Height = 1;
143 resdesc.DepthOrArraySize = 1;
144 resdesc.MipLevels = 1;
145 resdesc.Format = DXGI_FORMAT_UNKNOWN;
146 resdesc.SampleDesc.Count = 1;
147 resdesc.SampleDesc.Quality = 0;
148 resdesc.Flags = D3D12_RESOURCE_FLAG_NONE;
149 resdesc.Layout = D3D12_TEXTURE_LAYOUT_ROW_MAJOR;
150
151 device.GetDevice()->CreateCommittedResource(&heapprop, D3D12_HEAP_FLAG_NONE, &resdesc, D3D12_RESOURCE_STATE_GENERIC_READ,
152 nullptr, IID_PPV_ARGS(m_constantBuffers[index].ReleaseAndGetAddressOf()));
153
154 auto basicHeapHandle = m_constantDescHeap->GetCPUDescriptorHandleForHeapStart();
155 basicHeapHandle.ptr += m_descriptorHeapIncrementSize * (index * 2 + 1);
156
157 D3D12_CONSTANT_BUFFER_VIEW_DESC cbvDesc = {};
158 cbvDesc.BufferLocation = m_constantBuffers[index]->GetGPUVirtualAddress();
159 cbvDesc.SizeInBytes = static_cast<UINT>(m_constantBuffers[index]->GetDesc().Width);
160
161 device.GetDevice()->CreateConstantBufferView(&cbvDesc, basicHeapHandle);
162
163 m_constantBuffers[index]->Map(0, nullptr, (void**)&m_pMapMatrix[index]);
164 }

66 void MRMapData::HitWithWall(MRTransform& transform, const float size)
67 {
68 auto hitToChip = [&](float chipX, float chipY) {
69 auto setU = [&]() {
70 transform.SetY(chipY + m_chipSize / 2 + size);
71 transform.SetVelocityY(0.0f);
72 };
73 auto setD = [&]() {
74 transform.SetY(chipY - m_chipSize / 2 - size);
75 transform.SetVelocityY(0.0f);
76 };
77 auto setR = [&]() {
78 transform.SetX(chipX - m_chipSize / 2 - size);
79 transform.SetVelocityX(0.0f);
80 };
81 auto setL = [&]() {
82 transform.SetX(chipX + m_chipSize / 2 + size);
83 transform.SetVelocityX(0.0f);
84 };
85
86 float nowX = transform.GetX();
87 float nowY = transform.GetY();
88 float nextX = transform.GetNextX();
89 float nextY = transform.GetNextY();
90
91 double max = (nowX + size) - (chipX - m_chipSize / 2),
92 may = (nowY + size) - (chipY - m_chipSize / 2);
93 double nax = (chipX + m_chipSize / 2) - (nowX - size),
94 nay = (chipY + m_chipSize / 2) - (nowY - size);
95 double _max = (nextX + size) - (chipX - m_chipSize / 2),
96 _may = (nextY + size) - (chipY - m_chipSize / 2);
97 double _nax = (chipX + m_chipSize / 2) - (nextX - size),
98 _nay = (chipY + m_chipSize / 2) - (nextY - size);
99 bool U = false, D = false, R = false, L = false;
100
101 //カド同士の判定
102 if (0 >= max && 0 >= may && _max > 0 && _may > 0)
103 {
104 if (_max >= _may)
105 {
106 //下にある
107 D = true;
108 setD();
109 }
110 else
111 {
112 //右にある
113 R = true;
114 setR();
115 }
116 }
117 if (0 >= nax && 0 >= may && _nax > 0 && _may > 0)
118 {
119 if (_nax >= _may)
120 {
121 //下にある
122 D = true;
123 setD();
124 }
125 else
126 {
127 //左にある
128 L = true;
129 setL();
130 }
131 }
132 if (0 >= max && 0 >= nay && _max > 0 && _nay > 0)
133 {
134 if (_max >= _nay)
135 {
136 //上にある
137 U = true;
138 setU();
139 }
140 else
141 {
142 //右にある
143 R = true;
144 setR();
145 }
146 }
147 if (0 >= nax && 0 >= nay && _nax > 0 && _nay > 0)
148 {
149 if (_nax >= _nay)
150 {
151 //上にある
152 U = true;
153 setU();
154 }
155 else
156 {
157 //左にある
158 L = true;
159 setL();
160 }
161 }
162
163 //上下左右の判定
164 if (may > 0 && nay > 0)
165 {
166 if (_nax > 0 &&
167 (nowX - size) > (chipX - m_chipSize / 2))
168 {
169 //左にある
170 L = true;
171 setL();
172 }
173 if (_max > 0 &&
174 (chipX + m_chipSize / 2) > (nowX + size))
175 {
176 //右にある
177 R = true;
178 setR();
179 }
180 }
181 if (max > 0 && nax > 0)
182 {
183 if (_nay > 0 &&
184 (nowY - size) > (chipY - m_chipSize / 2))
185 {
186 //上にある
187 U = true;
188 setU();
189 }
190 if (_may > 0 &&
191 (chipY + m_chipSize / 2) > (nowY + size))
192 {
193 //下にある
194 D = true;
195 setD();
196 }
197 }
198
199 };
200
201 int currentChipX = static_cast<int>((transform.GetX() - size) / m_chipSize);
202 int currentChipY = static_cast<int>((transform.GetY() - size) / m_chipSize);
203
204 for (int y = max(currentChipY - 3,0); y <= min(currentChipY + 3, m_height); ++y)
205 {
206 for (int x = max(currentChipX - 3, 0); x <= min(currentChipX + 3, m_width); ++x)
207 {
208 if (m_mapData[y][x] == 0) continue;
209 hitToChip(x * m_chipSize + m_chipSize * 0.5f, y * m_chipSize + m_chipSize * 0.5f);
210 }
211 }

257 void MRMapData::_DivideRooms()
258 {
259 const int room_margin = 3;
260 const int room_minimum = 6;
261 const int divide_margin = room_minimum + room_margin * 2 + 4;
262 const int radius_path = 1;
263 std::vector<ROOM_NODE> roomList;
264
265 if (m_width < divide_margin * 3 || m_height < divide_margin * 3)
266 {
267 OutputDebugString(L"map size too small");
268 return;
269 }
270
271 // 部屋の割り方を決める
272 int xNum = m_random.GetRanged(max(3, m_width / divide_margin - 3), m_width / divide_margin);
273 int yNum = m_random.GetRanged(max(3, m_height / divide_margin - 3), m_height / divide_margin);
274
275 uint32_t yetElement = xNum * yNum;
276 uint32_t yetRoom = static_cast<uint32_t>(yetElement * static_cast<float>(m_random.GetRanged(6, 8)) / 10.0f);
277 for (int y = 0; y < yNum; ++y)
278 {
279 for (int x = 0; x < xNum; ++x)
280 {
281 int u = 0;
282 if (m_random.GetRanged(0, yetElement) <= yetRoom)
283 {
284 u = 1;
285 --yetRoom;
286 }
287 else
288 {
289 u = 2;
290 }
291 --yetElement;
292 roomList.push_back(ROOM_NODE(m_width * x / xNum, m_height * y / yNum, m_width * (x + 1) / xNum, m_height * (y + 1) / yNum, u));
293 }
294 }
295 // subetenosetuzokujoukyouwohozonn
296 std::vector<std::set<int>> globalConnect(xNum * yNum);
297 // 部屋の繋がり方を決める
298 std::vector<int> connect(xNum * yNum, -1);
299 {
300 std::stack<int> st;
301 std::vector<int> reserve;
302 st.push(m_random.GetRanged(0, yNum * xNum - 1));
303 while (!st.empty())
304 {
305 int cur = st.top();
306 st.pop();
307 int x = cur % xNum;
308 int y = cur / xNum;
309
310 if (x - 1 >= 0 && connect[cur - 1] == -1)
311 {
312 reserve.push_back(cur - 1);
313 if (m_random.GetRanged(0, 0) == 0)
314 {
315 connect[cur - 1] = cur;
316 globalConnect[cur].insert(cur - 1);
317 globalConnect[cur - 1].insert(cur);
318 }
319 }
320 if (x + 1 < xNum && connect[cur + 1] == -1)
321 {
322 reserve.push_back(cur + 1);
323 if (m_random.GetRanged(0, 0) == 0)
324 {
325 connect[cur + 1] = cur;
326 globalConnect[cur].insert(cur + 1);
327 globalConnect[cur + 1].insert(cur);
328 }
329 }
330 if (y - 1 >= 0 && connect[cur - xNum] == -1)
331 {
332 reserve.push_back(cur - xNum);
333 if (m_random.GetRanged(0, 0) == 0)
334 {
335 connect[cur - xNum] = cur;
336 globalConnect[cur].insert(cur - xNum);
337 globalConnect[cur - xNum].insert(cur);
338 }
339 }
340 if (y + 1 < yNum && connect[cur + xNum] == -1)
341 {
342 reserve.push_back(cur + xNum);
343 if (m_random.GetRanged(0, 0) == 0)
344 {
345 connect[cur + xNum] = cur;
346 globalConnect[cur].insert(cur + xNum);
347 globalConnect[cur + xNum].insert(cur);
348 }
349 }
350 std::shuffle(reserve.begin(), reserve.end(), m_random.GetDevice());
351 for (auto& i : reserve)
352 {
353 st.push(i);
354 }
355 reserve.clear();
356 }
357 }
358
359 // マップを壁で埋める
360 for (int y = 0; y < m_height; ++y)
361 {
362 for (int x = 0; x < m_width; ++x)
363 {
364 m_mapData[y][x] = 1;
365 }
366 }
367
368 for (int i = 0; i < roomList.size(); ++i)
369 {
370 if (roomList[i].usedFor == 0)
371 {
372 continue;
373 }
374 if (roomList[i].usedFor == 1)
375 {
376 // 部屋の大きさを区画内でランダムに決める
377 int offsetX = m_random.GetRanged(roomList[i].topX + room_margin, roomList[i].bottomX - room_margin - room_minimum);
378 roomList[i].topX = offsetX;
379 roomList[i].bottomX = m_random.GetRanged(offsetX + room_minimum, roomList[i].bottomX - room_margin);
380
381 int offsetY = m_random.GetRanged(roomList[i].topY + room_margin, roomList[i].bottomY - room_margin - room_minimum);
382 roomList[i].topY = offsetY;
383 roomList[i].bottomY = m_random.GetRanged(offsetY + room_minimum, roomList[i].bottomY - room_margin);
384 }
385 if (roomList[i].usedFor == 2)
386 {
387 // 交差点の位置を点で打つ
388 int offsetX = m_random.GetRanged(roomList[i].topX + room_margin, roomList[i].bottomX - room_margin - room_minimum);
389 roomList[i].topX = offsetX;
390 roomList[i].bottomX = offsetX + 1;
391
392 int offsetY = m_random.GetRanged(roomList[i].topY + room_margin, roomList[i].bottomY - room_margin - room_minimum);
393 roomList[i].topY = offsetY;
394 roomList[i].bottomY = offsetY + 1;
395 }
396
397 // デバッグ用部屋番号
398 m_roomIndex.push_back(
399 ROOM_INDEX((roomList[i].bottomX + roomList[i].topX) / 2, (roomList[i].bottomY + roomList[i].topY) / 2, i));
400
401 // 部屋や交差点に当たる部分に穴をあける
402 for (int y = roomList[i].topY; y < roomList[i].bottomY; ++y)
403 {
404 for (int x = roomList[i].topX; x < roomList[i].bottomX; ++x)
405 {
406 m_mapData[y][x] = 0;
407 }
408 }
409 }

411 auto fill3X3 = [&](int Y, int X) {
412 for (int y = max(0, Y - radius_path); y <= min(Y + radius_path, m_height - 1); ++y)
413 {
414 for (int x = max(0, X - radius_path); x <= min(X + radius_path, m_width - 1); ++x)
415 {
416 m_mapData[y][x] = 0;
417 }
418 }
419 };
420 auto makePathes = [&]() {
421 for (int i = 0; i < roomList.size(); ++i)
422 {
423 if (connect[i] == -1)continue;
424 // 通路に穴をあける
425 int fromX = i % xNum;
426 int fromY = i / xNum;
427 int toX = connect[i] % xNum;
428 int toY = connect[i] / xNum;
429
430 //接続先が横にある時
431 if (fromY == toY)
432 {
433 int wallX = (fromX > toX) ? m_random.GetRanged(roomList[connect[i]].bottomX + room_margin, roomList[i].topX - room_margin) :
434 m_random.GetRanged(roomList[i].bottomX - room_margin, roomList[connect[i]].topX + room_margin);
435 int beginY = (roomList[i].bottomY - roomList[i].topY == 1) ? roomList[i].topY :
436 m_random.GetRanged(roomList[i].topY + radius_path, roomList[i].bottomY - radius_path - 1);
437 int endY = (roomList[connect[i]].bottomY - roomList[connect[i]].topY == 1) ? roomList[connect[i]].topY :
438 m_random.GetRanged(roomList[connect[i]].topY + radius_path, roomList[connect[i]].bottomY - radius_path - 1);
439 for (int x = min(roomList[connect[i]].bottomX - radius_path, wallX); x < max(roomList[connect[i]].bottomX - radius_path, wallX); ++x)
440 {
441 fill3X3(endY, x);
442 }
443 for (int x = min(wallX, roomList[i].topX + radius_path); x < max(wallX, roomList[i].topX + radius_path); ++x)
444 {
445 fill3X3(beginY, x);
446 }
447 for (int y = min(beginY, endY); y <= max(beginY, endY); ++y)
448 {
449 fill3X3(y, wallX);
450 }
451 }
452 //接続先が縦にある時
453 if (fromX == toX)
454 {
455 int wallY = (fromY > toY) ? m_random.GetRanged(roomList[connect[i]].bottomY + room_margin, roomList[i].topY - room_margin) :
456 m_random.GetRanged(roomList[i].bottomY - room_margin, roomList[connect[i]].topY + room_margin);
457 int beginX = (roomList[i].bottomX - roomList[i].topX == 1) ? roomList[i].topX :
458 m_random.GetRanged(roomList[i].topX + radius_path, roomList[i].bottomX - radius_path - 1);
459 int endX = (roomList[connect[i]].bottomX - roomList[connect[i]].topX == 1) ? roomList[connect[i]].topX :
460 m_random.GetRanged(roomList[connect[i]].topX + radius_path, roomList[connect[i]].bottomX - radius_path - 1);
461 for (int y = min(roomList[connect[i]].bottomY - radius_path, wallY); y < max(roomList[connect[i]].bottomY - radius_path, wallY); ++y)
462 {
463 fill3X3(y, endX);
464 }
465 for (int y = min(wallY, roomList[i].topY + radius_path); y < max(wallY, roomList[i].topY + radius_path); ++y)
466 {
467 fill3X3(y, beginX);
468 }
469 for (int x = min(beginX, endX); x <= max(beginX, endX); ++x)
470 {
471 fill3X3(wallY, x);
472 }
473 }
474 }
475 };
476
477 makePathes();
478
479 for (int i = 0; i < roomList.size(); ++i)
480 {
481 int x = i % xNum;
482 int y = i / xNum;
483 std::vector<int> reserve;
484 if (x - 1 >= 0)
485 {
486 reserve.push_back(i - 1);
487 }
488 if (x + 1 < xNum)
489 {
490 reserve.push_back(i + 1);
491 }
492 if (y - 1 >= 0 && connect[i - xNum] == -1)
493 {
494 reserve.push_back(i - xNum);
495 }
496 if (y + 1 < yNum && connect[i + xNum] == -1)
497 {
498 reserve.push_back(i + xNum);
499 }
500 std::shuffle(reserve.begin(), reserve.end(), m_random.GetDevice());
501 for (auto& r : reserve)
502 {
503 if (connect[i] != r && connect[connect[r]] != i)
504 {
505 connect[i] = r;
506 break;
507 }
508 }
509 reserve.clear();
510 }
511
512 for (int i = 0; i < roomList.size(); ++i)
513 {
514 if (m_random.GetRanged(0, 7) != 0)
515 {
516 connect[i] = -1;
517 }
518 else
519 {
520 globalConnect[connect[i]].insert(i);
521 globalConnect[i].insert(connect[i]);
522 }
523 }
524
525 makePathes();
526

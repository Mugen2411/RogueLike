106 void MRMapData::_DivideRooms()
107 {
108 const int room_margin = 3;
109 const int room_minimum = 6;
110 const int divide_margin = room_minimum + room_margin * 2 + 4;
111 const int radius_path = 1;
112 std::vector<ROOM_NODE> roomList;
113
114 if (m_width < divide_margin * 3 || m_height < divide_margin * 3)
115 {
116 OutputDebugString(L"map size too small");
117 return;
118 }
119
120 // 部屋の割り方を決める
121 int xNum = m_random.GetRanged(max(3, m_width / divide_margin - 3), m_width / divide_margin);
122 int yNum = m_random.GetRanged(max(3, m_height / divide_margin - 3), m_height / divide_margin);
123
124 uint32_t yetElement = xNum * yNum;
125 uint32_t yetRoom = static_cast<uint32_t>(yetElement * static_cast<float>(m_random.GetRanged(6, 8)) / 10.0f);
126 for (int y = 0; y < yNum; ++y)
127 {
128 for (int x = 0; x < xNum; ++x)
129 {
130 int u = 0;
131 if (m_random.GetRanged(0, yetElement) <= yetRoom)
132 {
133 u = 1;
134 --yetRoom;
135 }
136 else
137 {
138 u = 2;
139 }
140 --yetElement;
141 roomList.push_back(ROOM_NODE(m_width * x / xNum, m_height * y / yNum, m_width * (x + 1) / xNum, m_height * (y + 1) / yNum, u));
142 }
143 }
144 // subetenosetuzokujoukyouwohozonn
145 std::vector<std::set<int>> globalConnect(xNum * yNum);
146 // 部屋の繋がり方を決める
147 std::vector<int> connect(xNum * yNum, -1);
148 {
149 std::stack<int> st;
150 std::vector<int> reserve;
151 st.push(m_random.GetRanged(0, yNum * xNum - 1));
152 while (!st.empty())
153 {
154 int cur = st.top();
155 st.pop();
156 int x = cur % xNum;
157 int y = cur / xNum;
158
159 if (x - 1 >= 0 && connect[cur - 1] == -1)
160 {
161 reserve.push_back(cur - 1);
162 if (m_random.GetRanged(0, 0) == 0)
163 {
164 connect[cur - 1] = cur;
165 globalConnect[cur].insert(cur - 1);
166 globalConnect[cur - 1].insert(cur);
167 }
168 }
169 if (x + 1 < xNum && connect[cur + 1] == -1)
170 {
171 reserve.push_back(cur + 1);
172 if (m_random.GetRanged(0, 0) == 0)
173 {
174 connect[cur + 1] = cur;
175 globalConnect[cur].insert(cur + 1);
176 globalConnect[cur + 1].insert(cur);
177 }
178 }
179 if (y - 1 >= 0 && connect[cur - xNum] == -1)
180 {
181 reserve.push_back(cur - xNum);
182 if (m_random.GetRanged(0, 0) == 0)
183 {
184 connect[cur - xNum] = cur;
185 globalConnect[cur].insert(cur - xNum);
186 globalConnect[cur - xNum].insert(cur);
187 }
188 }
189 if (y + 1 < yNum && connect[cur + xNum] == -1)
190 {
191 reserve.push_back(cur + xNum);
192 if (m_random.GetRanged(0, 0) == 0)
193 {
194 connect[cur + xNum] = cur;
195 globalConnect[cur].insert(cur + xNum);
196 globalConnect[cur + xNum].insert(cur);
197 }
198 }
199 std::shuffle(reserve.begin(), reserve.end(), m_random.GetDevice());
200 for (auto& i : reserve)
201 {
202 st.push(i);
203 }
204 reserve.clear();
205 }
206 }
207
208 // マップを壁で埋める
209 for (int y = 0; y < m_height; ++y)
210 {
211 for (int x = 0; x < m_width; ++x)
212 {
213 m_mapData[y][x] = 1;
214 }
215 }
216
217 for (int i = 0; i < roomList.size(); ++i)
218 {
219 if (roomList[i].usedFor == 0)
220 {
221 continue;
222 }
223 if (roomList[i].usedFor == 1)
224 {
225 // 部屋の大きさを区画内でランダムに決める
226 int offsetX = m_random.GetRanged(roomList[i].topX + room_margin, roomList[i].bottomX - room_margin - room_minimum);
227 roomList[i].topX = offsetX;
228 roomList[i].bottomX = m_random.GetRanged(offsetX + room_minimum, roomList[i].bottomX - room_margin);
229
230 int offsetY = m_random.GetRanged(roomList[i].topY + room_margin, roomList[i].bottomY - room_margin - room_minimum);
231 roomList[i].topY = offsetY;
232 roomList[i].bottomY = m_random.GetRanged(offsetY + room_minimum, roomList[i].bottomY - room_margin);
233 }
234 if (roomList[i].usedFor == 2)
235 {
236 // 交差点の位置を点で打つ
237 int offsetX = m_random.GetRanged(roomList[i].topX + room_margin, roomList[i].bottomX - room_margin - room_minimum);
238 roomList[i].topX = offsetX;
239 roomList[i].bottomX = offsetX + 1;
240
241 int offsetY = m_random.GetRanged(roomList[i].topY + room_margin, roomList[i].bottomY - room_margin - room_minimum);
242 roomList[i].topY = offsetY;
243 roomList[i].bottomY = offsetY + 1;
244 }
245
246 // デバッグ用部屋番号
247 m_roomIndex.push_back(
248 ROOM_INDEX((roomList[i].bottomX + roomList[i].topX) / 2, (roomList[i].bottomY + roomList[i].topY) / 2, i));
249
250 // 部屋や交差点に当たる部分に穴をあける
251 for (int y = roomList[i].topY; y < roomList[i].bottomY; ++y)
252 {
253 for (int x = roomList[i].topX; x < roomList[i].bottomX; ++x)
254 {
255 m_mapData[y][x] = 0;
256 }
257 }
258 }

260 auto fill3X3 = [&](int Y, int X) {
261 for (int y = max(0, Y - radius_path); y <= min(Y + radius_path, m_height - 1); ++y)
262 {
263 for (int x = max(0, X - radius_path); x <= min(X + radius_path, m_width - 1); ++x)
264 {
265 m_mapData[y][x] = 0;
266 }
267 }
268 };
269 auto makePathes = [&]() {
270 for (int i = 0; i < roomList.size(); ++i)
271 {
272 if (connect[i] == -1)continue;
273 // 通路に穴をあける
274 int fromX = i % xNum;
275 int fromY = i / xNum;
276 int toX = connect[i] % xNum;
277 int toY = connect[i] / xNum;
278
279 //接続先が横にある時
280 if (fromY == toY)
281 {
282 int wallX = (fromX > toX) ? m_random.GetRanged(roomList[connect[i]].bottomX + room_margin, roomList[i].topX - room_margin) :
283 m_random.GetRanged(roomList[i].bottomX - room_margin, roomList[connect[i]].topX + room_margin);
284 int beginY = (roomList[i].bottomY - roomList[i].topY == 1) ? roomList[i].topY :
285 m_random.GetRanged(roomList[i].topY + radius_path, roomList[i].bottomY - radius_path - 1);
286 int endY = (roomList[connect[i]].bottomY - roomList[connect[i]].topY == 1) ? roomList[connect[i]].topY :
287 m_random.GetRanged(roomList[connect[i]].topY + radius_path, roomList[connect[i]].bottomY - radius_path - 1);
288 for (int x = min(roomList[connect[i]].bottomX - radius_path, wallX); x < max(roomList[connect[i]].bottomX - radius_path, wallX); ++x)
289 {
290 fill3X3(endY, x);
291 }
292 for (int x = min(wallX, roomList[i].topX + radius_path); x < max(wallX, roomList[i].topX + radius_path); ++x)
293 {
294 fill3X3(beginY, x);
295 }
296 for (int y = min(beginY, endY); y <= max(beginY, endY); ++y)
297 {
298 fill3X3(y, wallX);
299 }
300 }
301 //接続先が縦にある時
302 if (fromX == toX)
303 {
304 int wallY = (fromY > toY) ? m_random.GetRanged(roomList[connect[i]].bottomY + room_margin, roomList[i].topY - room_margin) :
305 m_random.GetRanged(roomList[i].bottomY - room_margin, roomList[connect[i]].topY + room_margin);
306 int beginX = (roomList[i].bottomX - roomList[i].topX == 1) ? roomList[i].topX :
307 m_random.GetRanged(roomList[i].topX + radius_path, roomList[i].bottomX - radius_path - 1);
308 int endX = (roomList[connect[i]].bottomX - roomList[connect[i]].topX == 1) ? roomList[connect[i]].topX :
309 m_random.GetRanged(roomList[connect[i]].topX + radius_path, roomList[connect[i]].bottomX - radius_path - 1);
310 for (int y = min(roomList[connect[i]].bottomY - radius_path, wallY); y < max(roomList[connect[i]].bottomY - radius_path, wallY); ++y)
311 {
312 fill3X3(y, endX);
313 }
314 for (int y = min(wallY, roomList[i].topY + radius_path); y < max(wallY, roomList[i].topY + radius_path); ++y)
315 {
316 fill3X3(y, beginX);
317 }
318 for (int x = min(beginX, endX); x <= max(beginX, endX); ++x)
319 {
320 fill3X3(wallY, x);
321 }
322 }
323 }
324 };
325
326 makePathes();
327
328 for (int i = 0; i < roomList.size(); ++i)
329 {
330 int x = i % xNum;
331 int y = i / xNum;
332 std::vector<int> reserve;
333 if (x - 1 >= 0)
334 {
335 reserve.push_back(i - 1);
336 }
337 if (x + 1 < xNum)
338 {
339 reserve.push_back(i + 1);
340 }
341 if (y - 1 >= 0 && connect[i - xNum] == -1)
342 {
343 reserve.push_back(i - xNum);
344 }
345 if (y + 1 < yNum && connect[i + xNum] == -1)
346 {
347 reserve.push_back(i + xNum);
348 }
349 std::shuffle(reserve.begin(), reserve.end(), m_random.GetDevice());
350 for (auto& r : reserve)
351 {
352 if (connect[i] != r && connect[connect[r]] != i)
353 {
354 connect[i] = r;
355 break;
356 }
357 }
358 reserve.clear();
359 }
360
361 for (int i = 0; i < roomList.size(); ++i)
362 {
363 if (m_random.GetRanged(0, 7) != 0)
364 {
365 connect[i] = -1;
366 }
367 else
368 {
369 globalConnect[connect[i]].insert(i);
370 globalConnect[i].insert(connect[i]);
371 }
372 }
373
374 makePathes();
375 }
